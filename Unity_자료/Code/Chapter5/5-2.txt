using UnityEngine;
using System.Collections;

public class ArrowController : MonoBehaviour {
    void Start() {
    }

    void Update() {
        // 프레임마다 등속으로 낙하시킨다
        transform.Translate(0, -0.1f, 0);

        // 화면 밖으로 나오면 오브젝트를 파괴한다
        if(transform.position.y < -5.0f) {
            Destroy(gameObject);
        }
    }
}
