using UnityEngine;
using System.Collections;

public class CarController : MonoBehaviour {

    float speed = 0;

    void Start() {
    }

    void Update() {
        if(Input.GetMouseButtonDown(0)) {       // 마우스를 클릭하면
            this.speed = 0.2f;                  // 처음 속도를 설정
        }

        transform.Translate(this.speed, 0, 0); // 이동
        this.speed *= 0.98f;                   // 감속
    }
}
